#include<bits/stdc++.h>
using namespace std;

#define int long long
const int maxn=750;

int n;
int a[maxn+5];

namespace Prime
{
	const int at=5;
	const int a[at]={2,3,5,7,11};
	
	inline int Fpow(int x,int y,int mod)
	{
		int res=1;
		x%=mod;
		while(y)
		{
			if(y&1) res=res*x%mod;
			x=x*x%mod;
			y>>=1;
		}
		return res;
	}
	inline bool Check(int p)
	{
		if(p<2) return 0;
		int cnt=0;
		int x=p-1,y,z;
		while(!(x&1)) x>>=1,cnt++;
		for(int i=0;i<at;i++)
		{
			if(p==a[i]) return 1;
			y=Fpow(a[i],x,p);
			for(int j=0;j<cnt;j++)
			{
				z=y*y%p;
				if(z==1&&y!=1&&y!=p-1) return 0;
				y=z;
			}
			if(y!=1) return 0;
		}
		return 1;
	}
}

namespace Graph
{
	const int maxm=maxn*maxn*2;
	struct Edge{int v,nxt;};
	
	Edge e[maxm+5];
	int hd[maxn+5],et;
	bool vis[maxn+5];
	int mt[maxn+5];
	bool co[maxn+5];
	
	inline void Init()
	{
		et=0;
		memset(mt,0,sizeof(mt));
		memset(e,-1,sizeof(e));
		memset(hd,-1,sizeof(hd));
	}
	inline void Adde(int u,int v)
	{
		e[et].v=v;
		e[et].nxt=hd[u],hd[u]=et++;
	}
	
	bool Dfs(int u)
	{
		int v;
		for(int i=hd[u];~i;i=e[i].nxt)
		{
			v=e[i].v;
			if(!vis[v])
			{
				vis[v]=1;
				if(!mt[v]||Dfs(mt[v]))
				{
					mt[u]=v,mt[v]=u;
					return 1;
				}
			}
		}
		return 0;
	}
	inline int Match()
	{
		int ans=0;
		for(int i=1;i<=n;i++)
		{
			if(co[i]) continue;
			memset(vis,0,sizeof(vis));
			if(!mt[i]) ans+=Dfs(i);
		}
		return ans;
	}
}
using namespace Graph;

signed main()
{
	freopen("cooking.in","r",stdin);
	freopen("cooking.out","w",stdout);
	
	int T;
	bool flg;
	scanf("%lld",&T);
	while(T--)
	{
		flg=0;
		Init();
		scanf("%lld",&n);
		for(int i=1;i<=n;i++)
		{
			scanf("%lld",&a[i]);
			if(a[i]==1)
			{
				if(flg)
				{
					n--,i--;
					continue;
				}
				else flg=1;
			}
			co[i]=a[i]&1;
			for(int j=1;j<i;j++)
			{
//				cout<<a[i]+a[j]<<" "<<Prime::Check(a[i]+a[j])<<endl;
				if(Prime::Check(a[i]+a[j]))
					Adde(i,j),Adde(j,i);
			}
		}
		printf("%lld\n",n-Match());
	}
	
	return 0;
}
