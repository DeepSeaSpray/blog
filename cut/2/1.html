#include<bits/stdc++.h>
using namespace std;

const int maxn=1e4;
const int maxm=1e5;

int n,m;
struct Edge{int v,nxt;};
Edge e[maxm+5];
int hd[maxn+5],et;

int tot,cnt;
vector<int> ans[maxn+5];
bool vis[maxn+5],ins[maxn+5];
stack<int> st;
int dfn[maxn+5],low[maxn+5];
int b[maxn+5];

void Dfs(int u)
{
	dfn[u]=low[u]=++cnt;
	st.push(u);
	ins[u]=1;
	int v;
	for(int i=hd[u];~i;i=e[i].nxt)
	{
		v=e[i].v;
		if(!dfn[v]) Dfs(v),low[u]=min(low[u],low[v]);
		else if(ins[v]) low[u]=min(low[u],low[v]);
	}
	if(dfn[u]==low[u])
	{
		b[u]=++tot;
		ins[u]=0;
		ans[tot].push_back(u);
		while(st.top()!=u)
		{
			v=st.top();
			b[v]=tot;
			ins[v]=0;
			ans[tot].push_back(v);
			st.pop();
		}
		st.pop();
	}
}

inline void Adde(int u,int v)
{
	e[et].v=v;
	e[et].nxt=hd[u],hd[u]=et++;
}

signed main()
{
	int u,v;
	memset(e,-1,sizeof(e));
	memset(hd,-1,sizeof(hd));
	scanf("%d%d",&n,&m);
	while(m--) 
		scanf("%d%d",&u,&v),Adde(u,v);
	
	for(int i=1;i<=n;i++)
		if(!dfn[i])
			Dfs(i);
	
	printf("%d\n",tot);
	for(int i=1;i<=tot;i++)
		sort(ans[i].begin(),ans[i].end());
	
	for(int i=1;i<=n;i++)
	{
		if(vis[b[i]]) continue;
		vis[b[i]]=1;
		for(int j=0,tpj=ans[b[i]].size();j<tpj;j++)
			printf("%d ",ans[b[i]][j]);
		printf("\n");
	}
	
	return 0;
}
