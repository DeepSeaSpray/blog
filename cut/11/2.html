#include<bits/stdc++.h>
using namespace std;

#define pii pair<int,int>
#define fi first
#define se second
#define INF INT_MAX
const int maxn=2*1e3;

int n,m;
int h[maxn+5];
int l[maxn+5],r[maxn+5],p[maxn+5],w[maxn+5];
int f[maxn+5][maxn+5];
pii lb[maxn+5],rb[maxn+5];
int lo[maxn+5],ro[maxn+5];
priority_queue<pii,vector<pii>,greater<pii>> ql[maxn+5],qr[maxn+5];

inline bool cmpl(int x,int y){return l[x]<l[y];}
inline bool cmpr(int x,int y){return r[x]>r[y];}

inline int Fun(priority_queue<pii,vector<pii>,greater<pii>> &q
	,int il,int ir,int hl,int hr)
{
	pii tp;
	while(!q.empty())
	{
		tp=q.top();
		if(p[tp.se]<il||p[tp.se]>ir||r[tp.se]<hl||l[tp.se]>hr)
			q.pop();
		else return tp.fi;
	}
	return INF;
}

signed main()
{
	freopen("lantern.in","r",stdin);
	freopen("lantern.out","w",stdout);
	
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++) scanf("%d",&h[i]);
	for(int i=1;i<=m;i++) scanf("%d%d%d%d",&p[i],&w[i],&l[i],&r[i]);
	
	for(int i=1;i<=m;i++) lo[i]=i;
	sort(lo+1,lo+m+1,cmpl);
	for(int i=1;i<=m;i++) ro[i]=i;
	sort(ro+1,ro+m+1,cmpr);
	
	for(int i=1;i<=m;i++)
	{
		for(int &j=lb[i].fi=p[i]+1;j>1&&h[j-1]>=l[i];j--);
		for(int &j=lb[i].se=p[i]-1;j<n&&h[j+1]>=l[i];j++);
		for(int &j=rb[i].fi=p[i]+1;j>1&&h[j-1]<=r[i];j--);
		for(int &j=rb[i].se=p[i]-1;j<n&&h[j+1]<=r[i];j++);
	}
	
	int x,y;
	for(int i=1;i<=m;i++)
	{
		for(int j=1;j<=m;j++)
		{
			x=lo[i],y=ro[j];
			f[x][y]=INF;
			int ll=max(lb[x].fi,rb[y].fi),rr=min(lb[x].se,rb[y].se);
			if(p[x]<ll||p[x]>rr||p[y]<ll||p[y]>rr
				||(r[y]<r[x]&&l[x]>l[y])) continue;
			if(ll==1&&rr==n) f[x][y]=0;
			else if(r[y]<r[x]) f[x][y]=f[x][x];
			else if(l[x]>l[y]) f[x][y]=f[y][y];
			else
			{
				f[x][y]=min({f[x][y]
					,Fun(ql[x],ll,rr,l[x],r[y])
					,Fun(qr[y],ll,rr,l[x],r[y])});
			}
			if(f[x][y]<INF)
			{
				ql[x].push(pii(f[x][y]+w[y],y));
				qr[y].push(pii(f[x][y]+w[x],x));
			}
		}
	}
	
	for(int i=1;i<=m;i++)
	{
		if(f[i][i]<INF) printf("%d\n",f[i][i]+w[i]);
		else printf("%d\n",-1);
	}
	
	return 0;
}
