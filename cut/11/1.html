#include<bits/stdc++.h>
using namespace std;

#define int long long 
const int mod=998244353;
const int maxn=2*1e6;

int n,m;
char s[maxn+5];
bool a[maxn+5];

int p[maxn+5];
int sm[maxn+5];

namespace M
{
	int fac[maxn+5];
	int inv[maxn+5];
	int pw[maxn+5];
	
	inline int fpow(int x,int y)
	{
		int res=1;
		x%=mod;
		while(y)
		{
			if(y&1) res=res*x%mod;
			x=x*x%mod;
			y>>=1;
		}
		return res;
	}
	inline void Init()
	{
		pw[0]=fac[0]=1;
		for(int i=1;i<=n;i++)
			fac[i]=fac[i-1]*i%mod,pw[i]=pw[i-1]*2%mod;
		inv[n]=fpow(fac[n],mod-2);
		for(int i=n-1;i>=0;i--) inv[i]=inv[i+1]*(i+1)%mod;
	}
	inline int C(int x,int y)
	{
		return fac[x]*inv[x-y]%mod*inv[y]%mod;
	}
}

using namespace M;

inline int Hs(int l,int r)
{
	return ((p[r]-p[l-1]*pw[r-l+1])%mod+mod)%mod;
}
inline int Fun(int x,int y,int len)
{
	int l=1,r=len,mid,ans=0;
	while(l<=r)
	{
		mid=(l+r)>>1;
		if(Hs(x,x+mid-1)==Hs(y,y+mid-1)) l=mid+1,ans=mid;
		else r=mid-1;
	}
	return ans;
}

signed main()
{
//	freopen("divide.in","r",stdin);
//	freopen("divide.out","w",stdout);
	
	scanf("%lld%lld",&n,&m);
	Init();
	scanf("%s",s+1);
	for(int i=1;i<=n;i++)
	{
		a[i]=s[i]-'0';
		p[i]=(p[i-1]*2%mod+a[i])%mod;
		sm[i]=sm[i-1]+a[i];
	}
	
	int ans=0;
	if(!sm[n])
	{
		for(int i=m;i<=n;i++) ans=(ans+C(n-1,i-1));
		printf("%lld %lld",0ll,ans);
	}
	else if(n==m)
	{
		printf("%lld %lld",sm[n],1ll);
	}
	else if(!sm[m])
	{
		int fir=1;
		while(!a[fir]) fir++;
		for(int i=m;i<=fir;i++) ans=(ans+C(fir-1,i-1))%mod;
		printf("%lld %lld",p[n],ans);
	}
	else
	{
		int len=n-m+1;
		int mx=1;
		ans=1;
		for(int i=2;i+len-1<=n;i++)
		{
			int mt=Fun(mx,i,len);
			if(mt==len) ans++;
			else if(a[mx+mt]<a[i+mt]) mx=i,ans=1;
		}
		if(a[mx+len-1])
		{
			int tmp=((Hs(mx,mx+len-1)-1)%mod+mod)%mod;//Real
			for(int i=1;i+len-1<=n;i++)
				if(tmp==Hs(i,i+len-1))
					ans++;
		}
//		printf("%lld\n",mx);
		printf("%lld %lld"
			,Hs(mx,mx+len-1)+sm[mx-1]+sm[n]-sm[mx+len-1]
			,ans);
	}
	
	return 0;
}
