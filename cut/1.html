#include<bits/stdc++.h>
using namespace std;

const int maxn=1e6;
const int maxm=1e5;
const int maxs=1<<10;
const int maxk=10;

int n,m,q;
int a[maxn+5];
int op[maxn+5],il[maxn+5],ir[maxn+5],ix[maxn+5];

int id[maxm+5],idt;
bool f[maxs+5][maxm+5];

struct Node{int l,r,s,lz;};
Node T[(maxn<<2)+5];

inline void Insert(int x)
{
	if(~id[x]) return;
	id[x]=idt++;
	for(int ist=1<<(idt-1),ied=1<<idt,i=ist;i<ied;i++)
	{
		for(int j=0;j<=m;j++) f[i][j]=f[i^ist][j];
		for(int j=x;j<=m;j++) f[i][j]|=f[i][j-x];
	}
}

inline void Pushup(int x){T[x].s=T[x<<1].s|T[x<<1|1].s;}
inline void Update(int x,int val){T[x].lz=val,T[x].s=1<<val;}
inline void Pushdown(int x)
{if(~T[x].lz)Update(x<<1,T[x].lz),Update(x<<1|1,T[x].lz),T[x].lz=-1;}

void Build(int x,int l,int r)
{
	T[x].l=l,T[x].r=r,T[x].lz=-1;
	if(l==r) T[x].s=1<<a[l];
	else
	{
		int mid=(r-l)/2+l;
		Build(x<<1,l,mid);
		Build(x<<1|1,mid+1,r);
		Pushup(x);
	}
}
void Modify(int x,int l,int r,int val)
{
	if(l<=T[x].l&&T[x].r<=r) Update(x,val);
	else
	{
		Pushdown(x);
		int mid=(T[x].r-T[x].l)/2+T[x].l;
		if(l<=mid) Modify(x<<1,l,r,val);
		if(mid<r) Modify(x<<1|1,l,r,val);
		Pushup(x);
	}
}
int Query(int x,int l,int r)
{
	if(l<=T[x].l&&T[x].r<=r) return T[x].s;
	else
	{
		Pushdown(x);
		int mid=(T[x].r-T[x].l)/2+T[x].l,tmp=0;
		if(l<=mid) tmp|=Query(x<<1,l,r);
		if(mid<r) tmp|=Query(x<<1|1,l,r);
		return tmp;
	}
}

signed main()
{
//	freopen("past.in","r",stdin);
//	freopen("past.out","w",stdout);

	memset(id,-1,sizeof(id));
	f[0][0]=1;

	scanf("%d%d%d",&n,&m,&q);
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]),Insert(a[i]),a[i]=id[a[i]];
	for(int i=1;i<=q;i++)
	{
		scanf("%d%d%d%d",&op[i],&il[i],&ir[i],&ix[i]);
		if(op[i]==1) Insert(ix[i]),ix[i]=id[ix[i]];
	}

	Build(1,1,n);

	for(int i=1;i<=q;i++)
	{
		if(op[i]==1) Modify(1,il[i],ir[i],ix[i]);
		else printf("%s\n",f[Query(1,il[i],ir[i])][ix[i]]?"Yes":"No");
	}

	return 0;
}
