#include<bits/stdc++.h>
using namespace std;

#define int long long
#define pii pair<int,int>
#define fi first
#define se second
const int maxn=4*1e5;

struct Graph
{
	struct Edge{int u,v,w,pt,nxt;};
	Edge e[maxn+5];
	int hd[maxn+5],et;
	inline void Init()
	{
		memset(e,-1,sizeof(e));
		memset(hd,-1,sizeof(hd));
	}
	inline void adde(int u,int v,int w)
	{
		e[et].u=u,e[et].v=v,e[et].w=w;
		e[et].nxt=hd[u],hd[u]=et++;
	}
};

int n,m;
int a[maxn+5],asum;
Graph OG,NG;

stack<int> st;
int dfn[maxn+5],dt;
int lst[maxn+5],low[maxn+5];
int vt;
vector<int> ve[maxn+5];

int ans;
int bel[maxn+5];

map<int,map<int,int> > mp;
vector<pii> vet;

inline void Insert(int id,int u)
{
	bel[u]=id;
	ve[id].push_back(u);
	if(lst[u])
	{
		NG.e[NG.et].pt=u;
		NG.adde(id,lst[u],0);
		NG.e[NG.et].pt=u;
		NG.adde(lst[u],id,0);
	}
	lst[u]=id;
}

void Tarjan(int u,int fa)
{
	dfn[u]=low[u]=++dt;
	st.push(u);
	int v,ct=0;
	for(int i=OG.hd[u];~i;i=OG.e[i].nxt)
	{
		v=OG.e[i].v;
		if(!dfn[v])
		{
			ct++;
			Tarjan(v,u);
			low[u]=min(low[u],low[v]);
			if(low[v]>=dfn[u])
			{
				vt++;
				Insert(vt,u);
				while(st.top()!=v)
					Insert(vt,st.top()),st.pop();
				Insert(vt,v),st.pop();
			}
		}
		else if(v!=fa) low[u]=min(low[u],dfn[v]);
	}
}

inline void Getans(int id,int pt)
{
	int u,v,ttmp;
	if(ve[id].size()==2)
	{
		u=ve[id][0],v=ve[id][1];
		ans+=abs(a[v])*mp[u][v];
		a[u]+=a[v];
		return;
	}
	int tsum=0,wsum;
	for(int i=0,itp=ve[id].size();i<itp;i++)
		tsum+=a[ve[id][i]];
	ttmp=tsum;
	a[pt]-=tsum;
	tsum=0,wsum=0;
	vet.clear();
	for(int i=0,itp=ve[id].size();i<itp;i++)
	{
		int j=(i+1)%itp;
		u=ve[id][i],v=ve[id][j];
		tsum+=a[u];
		wsum+=mp[u][v];
		vet.push_back(pii(tsum,mp[u][v]));
	}
	sort(vet.begin(),vet.end());
	int ssum=0,tmp=-1;
	for(int i=0,itp=vet.size();i<itp;i++)
	{
		ssum+=vet[i].se;
		if(ssum>wsum/2)
		{
			tmp=vet[i].fi;
			break;
		}
	}
	for(int i=0,itp=vet.size();i<itp;i++)
		ans+=abs(vet[i].fi-tmp)*vet[i].se;
	a[pt]=ttmp;
}

void Dfs(int u,int fa,int pt)
{
	int v;
	for(int i=NG.hd[u];~i;i=NG.e[i].nxt)
	{
		v=NG.e[i].v;
		if(v==fa) continue;
		Dfs(v,u,NG.e[i].pt);
	}
	Getans(u,pt);
}

signed main()
{
	freopen("transfer.in","r",stdin);
	freopen("transfer.out","w",stdout);
	
	OG.Init();NG.Init();
	scanf("%lld%lld",&n,&m);
	for(int i=1;i<=n;i++)
		scanf("%lld",&a[i]),asum+=a[i];
	asum/=n;
	for(int i=1;i<=n;i++)
		a[i]-=asum;
	int iu,iv,iw;
	for(int i=1;i<=m;i++)
	{
		scanf("%lld%lld%lld",&iu,&iv,&iw);
		mp[iu][iv]=mp[iv][iu]=iw;
		OG.adde(iu,iv,iw);
		OG.adde(iv,iu,iw);
	}
	
	Tarjan(1,0);
	
	Dfs(bel[1],0,0);
	
	printf("%lld",ans);
	
	return 0;
}
