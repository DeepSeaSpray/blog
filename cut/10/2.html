#include<bits/stdc++.h>
using namespace std;

const int maxn=1e5;
const int maxm=2*1e5;

int n,m;
struct Edge{int v,nxt;};
Edge e[maxm+5];
int hd[maxn+5],et;

int mn;
int de[maxn+5];
int dis[maxn+5];

inline void Adde(int u,int v)
{
	e[et].v=v;
	e[et].nxt=hd[u],hd[u]=et++;
}

void Dfs(int u,int fa)
{
	int v;
	if(de[u]>=de[mn]) mn=u;
	for(int i=hd[u];~i;i=e[i].nxt)
	{
		v=e[i].v;
		if(v==fa) continue;
		de[v]=de[u]+1;
		Dfs(v,u);
	}
}

signed main()
{
	freopen("tree.in","r",stdin);
	freopen("tree.out","w",stdout);
	
	int u,v;
	memset(e,-1,sizeof(e));
	memset(hd,-1,sizeof(hd));
	
	scanf("%d%d",&n,&m);
	for(int i=1;i<n;i++)
	{
		scanf("%d%d",&u,&v);
		Adde(u,v),Adde(v,u);
	}
	
	Dfs(1,0);
	
	int t1=mn,t2;
	memset(de,0,sizeof(de)),mn=0;
	Dfs(t1,0);
	
	for(int i=1;i<=n;i++) dis[i]=de[i];
	
	t2=mn;
	memset(de,0,sizeof(de)),mn=0;
	Dfs(t2,0);
	
	int t3=de[t1];
	int mid;
	if(!(t3&1))
	{
		t3/=2;
		for(int i=1;i<=n;i++)
			if(de[i]==t3&&dis[i]==t3) mid=i;
	}
	
//	printf("%d %d\n",t3,mid);
	while(m--)
	{
		scanf("%d",&u);
		if(u==mid) printf("Bob\n");
		else printf("Alice\n");
	}
	
	return 0;
}
