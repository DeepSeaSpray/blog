#include<bits/stdc++.h>
using namespace std;

#define int long long
const int maxn=1e7;
const int maxm=1e6;
const int maxs=50;

int n;

int pr[maxm+5],prt;
int mn[maxn+5];
bool vis[maxn+5];

int ans[maxn+5];

int st[maxs+5],stt;

inline void Write(int x)
{
	stt=0;
	while(x)
	{
		st[stt++]=x%10;
		x/=10;
	}
	while(stt--) putchar(st[stt]+'0');
}

signed main()
{
	freopen("factorial.in","r",stdin);
	freopen("factorial.out","w",stdout);
	
	scanf("%lld",&n);
	
	int i,j,k;
	
	for(i=2;i<=n;i++)
	{
		if(!vis[i]) pr[prt++]=i,mn[i]=i;
		
		if(mn[i]!=0)
		{
			j=n/i;
			k=j*(n+1)-i*(j+1)*j/2;
			ans[mn[i]]+=k;
		}
		if(i*mn[i]<=n) mn[i*mn[i]]=mn[i];
		
		for(j=0;j<prt&&i*pr[j]<=n;j++)
		{
			vis[i*pr[j]]=1;
			if(i%pr[j]==0) break;
		}
	}
	
	putchar('f');putchar('(');
	Write(n);
	putchar(')');putchar('=');
	bool flg=0;
	for(int i=2;i<=n;i++)
	{
		if(ans[i]==0) continue;
		if(flg) putchar('*');
		flg=1;
		Write(i);
		if(ans[i]>1) putchar('^'),Write(ans[i]);
	}
	
	return 0;
}
