#include<bits/stdc++.h>
using namespace std;

const int maxn=3*1e3;

int n,m,K,B;
char s[maxn+5][maxn+5];

bool vis[maxn+5][maxn+5];
int ans,res;

void Dfs(int x,int y)
{
	if(vis[x][y]) return;
	vis[x][y]=1;
	if(s[x][y]=='k') res++;
	else if(s[x][y]=='b')
	{
		for(int i=1;i<=n;i++)
			Dfs(i,y);
		for(int i=1;i<=m;i++)
			Dfs(x,i);
	}
}

signed main()
{
	freopen("boom.in","r",stdin);
	freopen("boom.out","w",stdout);
	
	scanf("%d%d%d%d",&n,&m,&K,&B);
	for(int i=1;i<=n;i++)
		scanf("%s",s[i]+1);
	
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			if(s[i][j]=='b')
			{
				memset(vis,0,sizeof(vis));
				vis[i][j]=1,res=0;
				for(int k=1;k<=m;k++)
					if(s[i][k]=='b')
						Dfs(i,k);
				ans=max(ans,res);
				
				memset(vis,0,sizeof(vis));
				vis[i][j]=1,res=0;
				for(int k=1;k<=n;k++)
					if(s[k][j]=='b')
						Dfs(k,j);
				ans=max(ans,res);
			}
		}
	}
	
	printf("%d",ans);
	
	return 0;
}
