#include<bits/stdc++.h>
using namespace std;

#define pii pair<int,bool>
#define fi first
#define se second
const int maxn=3*1e3;
const int maxm=maxn*maxn;

int n,m,K,B;
char s[maxn+5][maxn+5];

int v[maxn+5][maxn+5];
int asum[maxn+5],bsum[maxn+5];
int alst[maxn+5],blst[maxn+5];

set<int> sex,sey;
set<int>::iterator itx,ity;
set<int> vex,vey;
int sum,res,mn;
bool rflg;

struct Edge{int v,nxt;};
Edge e[(maxm<<2)+5];
int hd[maxm+5],et;

int ans;
bool vis[maxm+5];

inline int Pt(int x,int y){return x*m+y;}
inline int Rpx(int x){return x/m;}
inline int Rpy(int x){return x%m;}

inline void Adde(int u,int v)
{
	e[et].v=v;
	e[et].nxt=hd[u],hd[u]=et++;
}
inline void Uadde(int u,int v){Adde(u,v),Adde(v,u);}

void Dfs(int u,int fa)
{
	if(sex.find(Rpx(u))==sex.end())
		sex.insert(Rpx(u)),sum+=asum[Rpx(u)];
	if(sey.find(Rpy(u))==sey.end())
		sey.insert(Rpy(u)),sum+=bsum[Rpy(u)];
	int v;
	bool flgx=0,flgy=0;
	vis[u]=1;
	for(int i=hd[u];~i;i=e[i].nxt)
	{
		v=e[i].v;
		if(v==fa) continue;
		if(vis[v])
		{
			continue;
			rflg=1;
		}
		Dfs(v,u);
		if(Rpx(u)==Rpx(v)) flgx=1;
		if(Rpy(u)==Rpy(v)) flgy=1;
	}
	if(Rpx(u)==Rpx(fa)) flgx=1;
	if(Rpy(u)==Rpy(fa)) flgy=1;
	if(!flgx) vex.insert(Rpx(u));
	if(!flgy) vey.insert(Rpy(u));
}

signed main()
{
	freopen("boom.in","r",stdin);
	freopen("boom.out","w",stdout);
	
	memset(e,-1,sizeof(e));
	memset(hd,-1,sizeof(hd));
	memset(alst,-1,sizeof(alst));
	memset(blst,-1,sizeof(blst));
	
	scanf("%d%d%d%d",&n,&m,&K,&B);
	for(int i=0;i<n;i++)
	{
		scanf("%s",s[i]);
		for(int j=0;j<m;j++)
		{
			v[i][j]=s[i][j]=='k';
			asum[i]=asum[i]+v[i][j];
			bsum[j]=bsum[j]+v[i][j];
			if(s[i][j]=='b')
			{
				if(~alst[i]) Uadde(Pt(i,j),Pt(i,alst[i]));
				alst[i]=j;
				if(~blst[j]) Uadde(Pt(i,j),Pt(blst[j],j));
				blst[j]=i;
			}
		}
	}
	
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if(!vis[Pt(i,j)]&&s[i][j]=='b')
			{
				sum=0;
				rflg=0;
				mn=INT_MAX;
				sex.clear(),sey.clear();
				vex.clear(),vey.clear();
				Dfs(Pt(i,j),-1);
				for(itx=sex.begin();itx!=sex.end();itx++)
					for(ity=sey.begin();ity!=sey.end();ity++)
						sum-=v[*itx][*ity];
				if(!rflg)
				{
					for(itx=vex.begin();itx!=vex.end();itx++)
					{
						res=asum[*itx];
						for(ity=sey.begin();ity!=sey.end();ity++)
							res-=v[*itx][*ity];
						mn=min(mn,res);
					}
					for(ity=vey.begin();ity!=vey.end();ity++)
					{
						res=bsum[*ity];
						for(itx=sex.begin();itx!=sex.end();itx++)
							res-=v[*itx][*ity];
						mn=min(mn,res);
					}		
				}
				if(mn==INT_MAX) mn=0;
				ans=max(ans,sum-mn);
			}
		}
	}
	
	printf("%d",ans);
	
	return 0;
}
