#include<bits/stdc++.h>
using namespace std;

#define int long long
const int maxn=1e6;

int n;
int a[maxn+5];

int sum,ans,tmp,cnt;

signed main()
{
//	freopen("training4.in","r",stdin);
//	freopen("training.out","w",stdout);
	
	scanf("%lld",&n);
	for(int i=1;i<=n;i++)
		scanf("%lld",&a[i]);
	
	sort(a+1,a+n+1);
	
	for(int i=1;i<=n;i++)
	{
		sum+=a[i];
		tmp=floor(sum*1.0/i);
		cnt=a+i-upper_bound(a+1,a+i+1,tmp)+1;
		ans=max(ans,cnt);
	}
	
	printf("%lld",ans);
	
	return 0;
}
