#include<bits/stdc++.h>
using namespace std;

#define int long long
const int maxn=2500;

int n,m;
struct Node{int l,r,w;};
Node a[maxn+5];

int f[maxn+5];
int sum,ans,res;

priority_queue<int,vector<int>,greater<int> > qu;

inline bool cmp(Node x,Node y){return x.r<y.r||(x.r==y.r&&x.l<y.l);}

signed main()
{
//	freopen("responsible.in","r",stdin);
//	freopen("responsible.out","w",stdout);
	
	scanf("%lld%lld",&n,&m);
	for(int i=1;i<=n;i++)
		scanf("%lld%lld%lld",&a[i].l,&a[i].r,&a[i].w),sum+=a[i].w;
	
	sort(a+1,a+n+1,cmp);
	
	for(int i=0;i<=n;i++)
	{
		ans=max(ans,f[i]);
		while(!qu.empty()) qu.pop();
		res=0;
		for(int j=i+1;j<=n;j++)
		{
			if(a[i].r<a[j].l)
			{
				qu.push(a[j].w);
				res+=a[j].w;
				if((int)qu.size()>m)
				{
					res-=qu.top();
					qu.pop();
				}
				f[j]=max(f[j],f[i]+res);
			}
		}
	}
	
	printf("%lld",sum-ans);
	
	return 0;
}
