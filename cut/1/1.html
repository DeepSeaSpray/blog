#include<bits/stdc++.h>
using namespace std;

const int maxn=1e6;

int l,r;

int phi[maxn+5];
int pr[maxn+5],prt;
bool vis[maxn+5];

signed main()
{
	scanf("%d%d",&l,&r);

	for(int i=2;i<=maxn;i++)
	{
		if(!vis[i]) pr[prt++]=i,phi[i]=i-1;
		for(int j=0;j<prt&&i*pr[j]<=maxn;j++)
		{
			vis[i*pr[j]]=1;
			if(i%pr[j]==0)
			{
				phi[i*pr[j]]=phi[i]*pr[j];
				break;
			}
			phi[i*pr[j]]=phi[i]*phi[pr[j]];
		}
	}

	for(int i=l;i<=r;i++)
		printf("%d\n",phi[i]);

	return 0;
}
